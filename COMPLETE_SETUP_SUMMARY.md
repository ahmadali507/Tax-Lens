## 📝 USAGE PATTERNS

### Pattern 1: Basic (Simplest)
```typescript
import { scrapedProjects } from '@/data/scraped-projects';

export default function ProjectsPage() {
    const projects = scrapedProjects;
}
```

### Pattern 2: With Filtering
```typescript
import ProjectDataService from '@/lib/services/project-data.service';

const ongoing = ProjectDataService.filterProjects(scrapedProjects, {
    status: 'ongoing'
});
```

### Pattern 3: With Statistics
```typescript
const stats = ProjectDataService.getStatistics(scrapedProjects);
console.log(stats.totalProjects);
console.log(stats.averageProgress);
```

### Pattern 4: Advanced Search
```typescript
const results = ProjectDataService.filterProjects(scrapedProjects, {
    status: 'ongoing',
    searchTerm: 'Karachi',
    minProgress: 50
});
```
